# Adding group 
- no users by default
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# groupadd engineering
[root@ceph-vipin-tfa-q5w08s-node6 ~]# grep engineering /etc/group
engineering:x:10004:
```

## groupadd -r will add system group as deifinde min and max rom /etc/login.defs file

# Modifying group name
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# groupmod -n engineers engineering
[root@ceph-vipin-tfa-q5w08s-node6 ~]# grep engineers /etc/group
engineers:x:10004:
[root@ceph-vipin-tfa-q5w08s-node6 ~]#
```

# Adding supplmentary greoup to the usser
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# groupmod -aU operator1 engineers
[root@ceph-vipin-tfa-q5w08s-node6 ~]# grep engineers /etc/group
engineers:x:10004:operator1
```

# Listing groups of a user
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# groups operator1
operator1 : operator1 engineers
[root@ceph-vipin-tfa-q5w08s-node6 ~]# id operator1
uid=10002(operator1) gid=10002(operator1) groups=10002(operator1),10004(engineers)
```

# overwriting supplementary group of user -G ( -g for primary group ) 
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# useradd devops
[root@ceph-vipin-tfa-q5w08s-node6 ~]# groups devops
devops : devops
[root@ceph-vipin-tfa-q5w08s-node6 ~]# usermod -G wheel devops
[root@ceph-vipin-tfa-q5w08s-node6 ~]# id devops
uid=10004(devops) gid=10005(devops) groups=10005(devops),10(wheel)
[root@ceph-vipin-tfa-q5w08s-node6 ~]# usermod -G engineers devops
[root@ceph-vipin-tfa-q5w08s-node6 ~]# id devops
uid=10004(devops) gid=10005(devops) groups=10005(devops),10004(engineers)
[root@ceph-vipin-tfa-q5w08s-node6 ~]#
```

# appending supplementary group to the user  -aG
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# usermod -aG wheel devops
[root@ceph-vipin-tfa-q5w08s-node6 ~]# id devops
uid=10004(devops) gid=10005(devops) groups=10005(devops),10(wheel),10004(engineers)
```
