# Adding user 
- -c description
- -s the shell for user
- -u user id
- -d the home directory 
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# useradd -c "Test user 123" -s /sbin/nologin -u 10000 -d /dev/null testuser123
useradd: warning: the home directory /dev/null already exists.
useradd: Not copying any file from skel directory into it.

[root@ceph-vipin-tfa-q5w08s-node6 ~]# cat /etc/passwd | grep testuser
testuser:x:1001:1001::/home/testuser:/bin/bash
testuser123:x:10000:10000:Test user 123:/dev/null:/sbin/nologin
```

# modifying user using usermod
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# usermod -c "Ceph user for automation" cephuser
[root@ceph-vipin-tfa-q5w08s-node6 ~]# grep cephuser /etc/passwd
cephuser:x:1000:1000:Ceph user for automation:/home/cephuser:/bin/bash
```
