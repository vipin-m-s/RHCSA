# There are 3 files created by default with FD of 0,1,2
- 0 = stdin
- 1 = stdout
- 2 = stderr

## by default 
- stdin = keyboard
- stdout = terminal
- stderr = terminal

# redirecting 
-  stdout to a file + overwrite
pwd > /tmp/out

-  stdout to a file + append
pwd >> /tmp/out

- stderr to a file + overwrite
pwd 2> /tmp/err

- stderr to a file + append
pwd 2>> /tmp/err

- stdout and stderr to the file
find / -name test > /tmp/out 2> /tmp/err

- redirect the err to same place where out is being redirected to 
find / -name test > /tmp/out 2> &1

# Pipes
- stdout of cmd1 -> stdin of cmd2
- stdout of cmd2 = terminal 
ls | wc -l

# Tee 
- print ls -ltr on screen + save the contents to /tmp/out
ls -ltr | tee /tmp/out

- to append and not overwrite
ls -ltr | tee -a /tmp/out
