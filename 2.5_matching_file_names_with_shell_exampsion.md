```
[ec2-user@ip-172-31-19-137 ~]$ ls b*
```

# double quote and single quote
Single quotes will not perform the exppansion
```
[root@ceph-vipin-tfa-q5w08s-node6 ~]# echo "the value of $USER is $USER"
the value of root is root

[root@ceph-vipin-tfa-q5w08s-node6 ~]# echo "the value of \$USER is ${USER}"
the value of $USER is root

[root@ceph-vipin-tfa-q5w08s-node6 ~]# echo 'the value of ${USER} is ${USER}'
the value of ${USER} is ${USER}
```
