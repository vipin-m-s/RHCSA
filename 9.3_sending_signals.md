# SIGTERM
- allows the process to complete task and to perform cleanup routines
```
[root@atollon-client ~]# ps aux | grep sleep
root       51338  0.0  0.0 226820  2108 pts/0    S    12:21   0:00 sleep 5m
root       51341  0.0  0.0 227688  2268 pts/0    S+   12:21   0:00 grep --color=auto sleep

[root@atollon-client ~]# kill 51338
```

# SIGKILL 
- terminate abruptly, process cannot run cleanup routines
```
[root@atollon-client ~]# ps aux | grep sleep
root       51349  0.0  0.0 226820  2064 pts/0    S    12:22   0:00 sleep 5m
root       51355  0.0  0.0 227688  2284 pts/0    S+   12:23   0:00 grep --color=auto sleep
Thu Apr 16 12:23:04 PM UTC 2026
[root@atollon-client ~]# kill -9 51349
Thu Apr 16 12:23:10 PM UTC 2026
[1]+  Killed                  sleep 5m
```

- kill using the name
```
[root@atollon-client ~]# sleep 5m &
[1] 51359
Thu Apr 16 12:23:41 PM UTC 2026
[root@atollon-client ~]# kill -SIGKILL 51359
Thu Apr 16 12:23:50 PM UTC 2026
[1]+  Killed                  sleep 5m
```

# pgrep 
- filter the process ID and command belonging to a user
```
pgrep -l -u devops
[devops@atollon-client ~]$ pgrep -l -u devops
51726 sleep
51729 bash
```

# pkill 
- kill all processes of a user
- This will exit even the bash shell openeed by the user
- 
```
[root@atollon-client ~]# pgrep -l -u devops
51726 sleep
Thu Apr 16 01:34:36 PM UTC 2026
[root@atollon-client ~]# pkill -SIGKILL -u devops
Thu Apr 16 01:34:47 PM UTC 2026
[root@atollon-client ~]# pgrep -l -u devops
Thu Apr 16 01:34:48 PM UTC 2026
[root@atollon-client ~]#
```

## If SIGNKILL is not passed, then bashshell survives but all other processes are killed
```
[devops@atollon-client ~]$ pgrep -l -u devops
51965 bash
51994 sleep
51995 sleep
51996 sleep


[root@atollon-client ~]# pkill -u devops
Thu Apr 16 01:39:12 PM UTC 2026


[devops@atollon-client ~]$ pgrep -l -u devops
51965 bash
[1]   Terminated              sleep 5m
[2]-  Terminated              sleep 10m
[3]+  Terminated              sleep 15m
```
