# SIGTERM
- allows the process to complete task and to perform cleanup routines
```
[root@atollon-client ~]# ps aux | grep sleep
root       51338  0.0  0.0 226820  2108 pts/0    S    12:21   0:00 sleep 5m
root       51341  0.0  0.0 227688  2268 pts/0    S+   12:21   0:00 grep --color=auto sleep

[root@atollon-client ~]# kill 51338
```

# SIGKILL 
- terminate abruptly, process cannot run cleanup routines
```
[root@atollon-client ~]# ps aux | grep sleep
root       51349  0.0  0.0 226820  2064 pts/0    S    12:22   0:00 sleep 5m
root       51355  0.0  0.0 227688  2284 pts/0    S+   12:23   0:00 grep --color=auto sleep
Thu Apr 16 12:23:04 PM UTC 2026
[root@atollon-client ~]# kill -9 51349
Thu Apr 16 12:23:10 PM UTC 2026
[1]+  Killed                  sleep 5m
```

- kill using the name
```
[root@atollon-client ~]# sleep 5m &
[1] 51359
Thu Apr 16 12:23:41 PM UTC 2026
[root@atollon-client ~]# kill -SIGKILL 51359
Thu Apr 16 12:23:50 PM UTC 2026
[1]+  Killed                  sleep 5m
```
